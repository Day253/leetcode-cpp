\section{Pow(x,n)} %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\subsubsection{描述}
Implement pow(x, n).


\subsubsection{分析}
二分法，$x^n = x^{n/2} \times x^{n/2} \times x^{n\%2}$


\subsubsection{代码}
\begin{Code}
//LeetCode, Pow(x, n)
// 二分法，$x^n = x^{n/2} * x^{n/2} * x^{n\%2}$
class Solution {
private:
    double power(double x, int n) {
        if (n == 0) return 1;
        double v = power(x, n / 2);
        if (n % 2 == 0) return v * v;
        else return v * v * x;
    }
public:
    double pow(double x, int n) {
        if (n < 0) return 1.0 / power(x, -n);
        else return power(x, n);
    }
};
\end{Code}
